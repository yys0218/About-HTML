변수(variable)

: 하나의 값만 저장하며, 그 값은 달라질 수 있음
< - > 상수 constant : 값을 한 번 지정하면 바뀌지 않는 데이터

변수 선언
- 사용이 유추 가능하게 이름짓기.
- 한 번 대입 후, 다른 값 대입 가능
- 재선언 가능
- 명명 규칙은 자바와 같음.

변수 선언과 사용
변수예약어 변수명;

< button type="button" id="var" > < /button > < script > var currentYear=2025;
var birthYear;
var age;

document.querySelector('#var').onclick=function(){
  birthYear = prompt("태어난 연도를 입력하세요 예:2000 ");
    if( birthYear === null || birthYear > "2025" ){
       alert("다시 입력해주세요");
    } else {
       alert("콘솔창을 확인해주세요.\n(ctrl+shift+j)");
       age = currentYear - birthYear;
       console.log("올해는 "+currentYear+"년 입니다.\n");
       console.log(birthYear+"년에 태어난 사람의 나이는 "+age+"살 입니다.");
    }
  }
</script>
